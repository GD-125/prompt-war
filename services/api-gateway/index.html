// deploy.sh — Key Commands
# 1. Enable APIs
gcloud services enable run.googleapis.com pubsub.googleapis.com firestore.googleapis.com

# 2. Build & push to Artifact Registry
docker build --platform linux/amd64 -t $REGION-docker.pkg.dev/$PROJECT/venueiq/api:latest services/api
docker push $REGION-docker.pkg.dev/$PROJECT/venueiq/api:latest

# 3. Deploy API service (public ingress)
gcloud run deploy venueiq-api \
  --image=$REGION-docker.pkg.dev/... \
  --min-instances=2 --max-instances=100 --concurrency=1000 \
  --cpu=2 --memory=1Gi --timeout=30s \
  --ingress=all --allow-unauthenticated --execution-environment=gen2

# 4. Deploy workers (internal ingress only)
gcloud run deploy venueiq-crowd-worker \
  --ingress=internal --no-allow-unauthenticated \
  --min-instances=1 --max-instances=50 --cpu=1 --memory=512Mi

# 5. Create Pub/Sub push subscription → worker
gcloud pubsub subscriptions create crowd-analysis-sub \
  --topic=venue-events \
  --push-endpoint=$WORKER_URL/pubsub/push \
  --push-auth-service-account=$PUBSUB_SA \
  --ack-deadline=60 --max-delivery-attempts=5 \
  --dead-letter-topic=venue-events-dead-letter

# 6. Blue/green traffic split for zero-downtime deploys
gcloud run services update-traffic venueiq-api \
  --to-revisions=LATEST=5,CURRENT=95 # canary
gcloud run services update-traffic venueiq-api --to-latest # promote
